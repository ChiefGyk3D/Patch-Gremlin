📝 QUICK MANUAL TESTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣  Test Notification (sends immediately)
   sudo systemctl start update-notifier.service
   → Check your Discord/Teams/Slack/Matrix for notification

2️⃣  Check Service Status
   sudo systemctl status update-notifier.service

3️⃣  View Logs
   sudo journalctl -u update-notifier.service -n 50

4️⃣  Check Timer Schedule
   sudo systemctl list-timers update-notifier.timer

5️⃣  Test Unattended Upgrades (Debian)
   sudo unattended-upgrades --debug --dry-run

6️⃣  Test DNF Automatic (RHEL)
   sudo dnf-automatic --downloadupdates


🔍 VERIFY INSTALLATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ls -la /etc/systemd/system/update-notifier.service
ls -la /etc/update-notifier/secrets.conf  # If using local mode


🎯 SUCCESS CRITERIA
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
✅ Test notification received on your platform
✅ Service status shows "succeeded"
✅ Logs show "HTTP 204" or "HTTP 200" (successful notification)
✅ Timer shows next scheduled run
✅ No errors in logs


🐛 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
If no notification received:
1. Check secrets: sudo cat /etc/update-notifier/secrets.conf
2. View detailed logs: sudo journalctl -u update-notifier.service -n 100
3. Test webhook URL manually with curl
4. Verify service environment: sudo systemctl show update-notifier.service
